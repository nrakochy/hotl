import { Card, CardGrid } from '@astrojs/starlight/components';

<CardGrid stagger>
	<Card title="A floor that never turns off" icon="approve-check">
		Every mutating tool call passes a permission gate, and `bash` runs under a
		kernel sandbox (Seatbelt on macOS, Landlock on Linux) that confines writes
		to the working directory. Writes to git hooks, shell rc, Makefiles, and
		agent-instruction files always stop and ask — in every mode.
		[Permissions & sandbox →](./permissions-and-sandbox/)
	</Card>
	<Card title="Nothing is ever lost" icon="document">
		Sessions are append-only event logs: `hotl resume` continues one,
		`hotl undo` reverses the agent's file changes via shadow-git snapshots,
		and context compaction adds a summary on top of history instead of
		rewriting it. [Overview →](./overview/)
	</Card>
	<Card title="Any model, one env var" icon="setting">
		`HOTL_MODEL=provider/model` — Anthropic's Messages API, or any
		OpenAI-compatible endpoint: OpenAI, Groq, a local Ollama server, or a
		gateway like LiteLLM. [Gateways & key sources →](./gateway/)
	</Card>
	<Card title="Standard protocols at the edges" icon="puzzle">
		An MCP client brings in external tools; `hotl acp` embeds the agent in any
		ACP-speaking editor; `hotl watch` supervises every agent in your tmux
		session. [MCP servers →](./mcp/)
	</Card>
</CardGrid>
